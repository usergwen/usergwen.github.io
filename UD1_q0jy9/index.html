driver.back()
driver.back()

使用AppiumDesktop录制一个可用的用例脚本

录制脚本如下：

# This sample code uses the Appium python client
# pip install Appium-Python-Client
# Then you can paste this into a file and simply run with Python

from appium import webdriver

caps = {}
caps["platformName"] = "android"
caps["deviceName"] = "emulator-5554"
caps["automationName"] = "uiautomator2"
caps["appPackage"] = "io.appium.android.apis"
caps["appActivity"] = ".ApiDemos"
caps["ensureWebviewsHavePages"] = True

driver = webdriver.Remote("http://localhost:4723/wd/hub", caps)

el1 = driver.find_element_by_accessibility_id("Access'ibility")
el1.click()
el2 = driver.find_element_by_accessibility_id("Custom View")
el2.click()
el3 = driver.find_element_by_xpath("/hierarchy/android.widget.FrameLayout/android.view.ViewGroup/android.widget.FrameLayout[2]/android.widget.LinearLayout/android.view.View[1]")
el3.click()
driver.back()
driver.back()
el4 = driver.find_element_by_accessibility_id("Text")
el4.click()
el5 = driver.find_element_by_accessibility_id("KeyEventText")
el5.click()
el6 = driver.find_element_by_id("io.appium.android.apis:id/text")
el6.click()
el6.send_keys("你好，Appium！")
el7 = driver.find_element_by_accessibility_id("Clear")
el7.click()

driver.quit()

上面这个脚本，只要执行流程就是，选择app中的文本框，输入一段文字，再清除所输入内容，退出app。
可以直接粘贴到pycharm或者vscode中执行一下，知道调整代码，能够正确执行上面的用例脚本。

重新组织录制脚本代码

function - init_app()

from appium import webdriver


def init_app():
    caps = {
        "platformName": "android",
        "deviceName": "emulator-5554",
        "automationName": "uiautomator2",
        "appPackage": "io.appium.android.apis",
        "appActivity": ".ApiDemos",
        "ensureWebviewsHavePages": True
    }
    driver = webdriver.Remote("http://127.0.0.1:4723/wd/hub", caps)
    return driver

funtion - click_el()

from appium import webdriver


def click_el(by, el):
    return driver.find_element(by, el).click()

function - send_key_to_el()

from appium import webdriver


def send_key_to_el(by, el, text):
    rs = driver.find_element(by, el)
    return rs.send_keys(text)

test_api_demo_text.py

from util.utils import click_el, driver, send_key_to_el


def test_api_demo_text():
    click_el("accessibility_id", "Access'ibility")
    click_el("accessibility_id", "Custom View")
    click_el("xpath",
             "/hierarchy/android.widget.FrameLayout/android.view.ViewGroup/android.widget.FrameLayout[2]/android.widget.LinearLayout/android.view.View[1]")
    driver.back()
    driver.back()
    click_el("accessibility_id", "Text")
    click_el("accessibility_id", "KeyEventText")
    send_key_to_el("id", "io.appium.android.apis:id/text", "你好，Appium！")
    click_el("accessibility_id", "Clear")

    driver.quit()
